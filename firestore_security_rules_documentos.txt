// Regras de segurança do Firestore para POPs e Manuais
// Cole estas regras no Firebase Console > Firestore Database > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Regra para a collection configuracoes/documentos
    match /configuracoes/documentos {
      // Permitir leitura para qualquer usuário autenticado
      allow read: if request.auth != null;
      
      // Permitir escrita apenas para usuários autenticados
      // TODO: Adicionar verificação de tipo de usuário (coordenador)
      allow write: if request.auth != null;
    }
    
    // MANTER AS REGRAS EXISTENTES DE OUTRAS COLLECTIONS:
    
    // Produtos
    match /produtos/{document=**} {
      allow read, write: if request.auth != null;
    }
    
    // Usuários
    match /usuarios/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth.uid == userId || request.auth != null;
    }
    
    // Equipamentos
    match /equipamentos/{document=**} {
      allow read, write: if request.auth != null;
    }
    
    // Reservas
    match /reservas/{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}
